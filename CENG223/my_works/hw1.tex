\begin{table}[ht]
\centering
\begin{tabular}{|l|l|l|l|} 
\hline

\hline
T    & F    & F    & T     \\

\end{tabular}

P(2) $\land$ Q(3) $\rightarrow$ P(2) $\land$ Q(2)\\
